def classIdentify(key, value, txt):
    if sum(val in txt for val in value) > 0:
        return(key)
    else:
        return (None)
    
def tempFinder(configData, txt):
    for key, value in configData.items():
        try:
            tempName = classIdentify(key, value, txt)
            if tempName is not None:
                break
        except:
            continue
    if tempName is not None:
        return (tempName)
    else:
        return ('NA')
